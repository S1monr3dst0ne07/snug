
"(Int a, Int b) => (Int c)
rout Math::Mul
{
    pull b;
    pull a;

    put c = 0;

    lab loop;
        jump done ~ a == 0;

        put c = c + b;
        put a = a + 1;

        jump loop;
    lab done;

    push c; 
}



"(Int a) => (Int !a)
rout Math::Fac
{
    pull n;

    save;
    push n - 1;
    sub Math::Fac;
    load;

    push n;
    sub Math::Mul;
}



rout main
{
    in 10;
    in 20;
    sub Math::Mul;
    out _;

    print _;
}
